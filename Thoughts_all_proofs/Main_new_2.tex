\[
\forall \text{ clauses } c_j, \quad \sum_{r=1}^3 G\bigl[\lfloor T_{k_{v_r}(l)}(x_{v_r}^j) \rfloor\bigr] \equiv 1 \pmod{3}.
\]
Verification runs in $O(n^3)$ time.
\end{theorem}

\begin{proof}
\textbf{1. Surface construction.} Set $l = 200\, n^3 \log_2 n + 1$. Partition into $m$ blocks of length $\ge 200\, n \log_2 n$, each with three subintervals. Assign cyclic twists and positions $x_v^j$. Separation and injectivity lemmas guarantee distinct indices.  
\textbf{2. Residue requirements.} Each clause needs one of 7 triples summing to 1 mod 3.  
\textbf{3. Existence.} Density lemma gives per-clause failure $<1/(100 n^3)$. Union bound yields a satisfying prefix.  
\textbf{4. Verification.} Compute modular sums in $O(n^3)$ time. Soundness and completeness hold by injectivity and density.
\end{proof}
